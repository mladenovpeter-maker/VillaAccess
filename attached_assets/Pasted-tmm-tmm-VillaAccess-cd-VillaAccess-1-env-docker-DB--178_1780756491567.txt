# 1. Вземи паролата от .env.docker и я приложи на DB ролята (без загуба на данни — през trust socket)
PW=$(grep '^POSTGRES_PASSWORD=' .env.docker | cut -d= -f2-)
docker compose exec -T postgres psql -U villa_user -d villa_access -c "ALTER USER villa_user PASSWORD '$PW';"

# 2. Рестартирай стека, за да хванат услугите новата (вече съвпадаща) парола
docker compose up -d
